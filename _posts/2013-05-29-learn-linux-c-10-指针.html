指针

    @1：字节存储，每个字节八位信息
    @2：可执行程序由
        代码 数据 
    @3：用数字表示地址，但是只指针变量
    @4：取地址运算符和简介寻址运算符
        取地址运算符：
            int *p, i; p = &i ; 相当于：
            int i ,*p = &i ;
        间接寻址运算符：
            p = &i
            *p 就是i的值，并且会改变i的值
    @5：指针赋值
        p = &i ;
        q = &j ;
        i = 1;
        *q = *p;  
    @6：指针作为实际参数，例子：
    

    #include <stdio.h>
    
    #define N 10
    
    static void max_min(int a[], int n, int *max, int *min) {
    	int i;
    	*max = *min = a[0];
    	for(i = 1; i < n; i++){
    		if (a[i] > *max) {
    			*max = a[i];
    		}
    		if (a[i] < *min) {
    			*min = a[i];
    		}
    	}
    }
    
    int main(void) {
    	int b[N], i, big, small;
    	printf("please enter %d number: \n", N);
    	for (i = 0; i < N; i++){
    		scanf("%d",&b[i]);
    	}
    	max_min(b, N, &big, &small);
    	printf("big:%d\n", big);
    	printf("small:%d\n", small);
    	return 0;
    }
    

    @7：const
    我青春，不改变
    @8：指针作为返回值
        ！永远不要返回自动局部变量的指针
        例子：
        

        #include <stdio.h>
        
        static int *max(int *a, int *b){
        	if(*a < *b)
        	  return a;
        	return b;
        }
        
        int main(void) {
        	int x, y;
        	x = 456;
        	y = 789;
        	printf("%d\n", *max(&x, &y));
        	return 0;
        }
        

    @9：结尾
        指针不总是和地址一样，比如用字而非字节划分内存的计算机
