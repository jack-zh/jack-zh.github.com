指针和数组

    @1：指针的算术运算
        只有指向相同数组的指针的比较才有意义
    @2：指针用于数组处理
        1:可以这样：
            #define N 10
            int a[N], sum, *p;
            sum = 0;
            for (p = &a[0]; p < &a[N]; p++)
                sum += *p;
            /* 注意：可以直接写a[N],不会检查 */
        2：一些操作：
            *p++ or *(p++)
            (*p)++
            *++p or *(++p)
            ++*p or ++(*p)
        3：修改1：
            #define N 10
            int a[N], sum, *p;
            sum = 0;
            while (p < &a[N])
                sum += *p++;
            /* 注意：可以直接写a[N],不会检查 */            
        4：修改栈的实现
            int *top_ptr = &contents[0];
            
            void push (int i){
                if (is_full())
                    stack_overflow();
                else
                    *top_ptr++ = i;
            }
            
            int pop (int i){
                if (is_empty())
                    stack_overflow();
                else
                    return *--top_ptr;/*切记 -- d的位置*/
            }

            例子：数组反向
            

            #include <stdio.h>
            
            #define N 10
            
            int main(void) {
                int a[N], *p;
                printf("Enter %d numbers: \n", N);
                for(p = a; p< a + N; p++)
                    scanf("%d", p);
                
                printf("In reverse order: ");
                for(p = a + N -1; p >= a; p--)
                    printf("%d", *p);  
                printf("\n");
                
                return 0;
            }

            #include <stdio.h>
            
            #define N 10
            
            void max_min(int a[], int *max, int *min);

            	int a[10];
            	int max, min;
            	int i;
            	printf("Enter 10 num:\n");
            	for (i = 0;i<N;i++)
            	  scanf("%d", &a[i]);
            	max_min(a, &max, &min);
            	printf("max:%d  min:%d", max, min);
            	return 0;
            }
            
            void max_min(int a[], int *max, int *min){
            	int i = 0;
            	*max = a[0];
            	*min = a[0];
            	for (i = 0; i< N; i++){
            		if(a[i] > *max)
            		  *max = a[i];
            		if(a[i] < *min)
            		  *min = a[i];
            	}
            }

        @5：指针和多维数组
            1：世界上本没有多维数组
            2：把多位数组当成一维数组
                int a[N][P], *p;
                for(p = &a[0][0]; p<= &a[N-1][P-1]; i++)
                {
                    *p = 0;
                }
            3：处理多维数组的行
                p = &a[i][0]
                or
                p = a[i]
                推算：
                    a[i] = *(a + i)
                    &a[i][0] = a[i] + 0
                    &a[i][0] = &(*(a[i] + 0))
                    &(*(a[i] + 0)) = a[i]
                    &a[i][0] = a[i]
            4：处理多位数组的行
                int a[N][P], *p, i;
                for(p=a[i]; p < = a[i] + P; p++)
                {
                    *p = 0;
                }
